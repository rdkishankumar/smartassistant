GitHub MCP Server enables:
- Create/manage repositories
- Commit and push code
- Create pull requests
- Manage issues

- All via natural language to LLM!

### **GitHub MCP Server Setup**

#### **Step 1: Create GitHub Token**

```
1. Go to GitHub Settings
   └─ Settings → Developer settings → Personal access tokens → Tokens (classic)

2. Create new token with scopes:
   ├─ repo (full control)
   ├─ read:user
   ├─ gist
   └─ workflow

3. Copy token (you'll need it next)
```

#### **Step 2: Update mcp-servers.json**

```json
{
  "mcpServers": {
    "filesystem": {
      "command": "npx",
      "args": [
        "-y",
        "@modelcontextprotocol/server-filesystem",
        "/Users/eazybytes/mcp"
      ],
      "env": {}
    },
    "github": {
      "command": "npx",
      "args": [
        "-y",
        "@modelcontextprotocol/server-github"
      ],
      "env": {
        "GITHUB_PERSONAL_ACCESS_TOKEN": "ghp_your_token_here"
      }
    }
  }
}
```

#### **Step 3: Secure Token Management**

```java
// Don't hardcode tokens!
```

`application.properties`:

### **Complete GitHub Integration Example**

```java
package com.eazybytes.mcpclient.controller;

import org.springframework.ai.chat.client.ChatClient;
import org.springframework.web.bind.annotation.*;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

/**
 * GitHub-enabled Chat Controller
 */
@RestController
@RequestMapping("/api/github-chat")
public class GitHubChatController {

    private static final Logger logger = 
        LoggerFactory.getLogger(GitHubChatController.class);

    private final ChatClient chatClient;

    public GitHubChatController(ChatClient.Builder chatClientBuilder) {
        this.chatClient = chatClientBuilder.build();
    }

    /**
     * Chat with GitHub tool access
     * 
     * Example: "Create a new repository called demo"
     */
    @PostMapping("/github-action")
    public ChatResponse performGitHubAction(@RequestBody ChatRequest request) {
        logger.info("GitHub action request: {}", request.getMessage());

        try {
            String response = chatClient.prompt()
                .user(request.getMessage())
                .call()
                .content();

            logger.info("GitHub action completed");
            return new ChatResponse(response);

        } catch (Exception e) {
            logger.error("GitHub action failed", e);
            return new ChatResponse("Error: " + e.getMessage());
        }
    }

    @lombok.Data
    @lombok.AllArgsConstructor
    @lombok.NoArgsConstructor
    public static class ChatRequest {
        private String message;
    }

    @lombok.Data
    @lombok.AllArgsConstructor
    public static class ChatResponse {
        private String response;
    }
}
```

### **Simple Custom MCP Server (Node.js)**

```javascript
// custom-mcp-server.js
const express = require('express');
const app = express();

app.use(express.json());

/**
 * List available tools
 */
app.get('/tools', (req, res) => {
    const tools = [
        {
            name: 'createTask',
            description: 'Create a new task in the system',
            inputSchema: {
                type: 'object',
                properties: {
                    title: { type: 'string' },
                    description: { type: 'string' },
                    priority: { 
                        type: 'string',
                        enum: ['low', 'medium', 'high']
                    }
                },
                required: ['title']
            }
        },
        {
            name: 'listTasks',
            description: 'List all tasks',
            inputSchema: {
                type: 'object',
                properties: {}
            }
        },
        {
            name: 'updateTask',
            description: 'Update an existing task',
            inputSchema: {
                type: 'object',
                properties: {
                    taskId: { type: 'string' },
                    status: { type: 'string' }
                },
                required: ['taskId']
            }
        }
    ];

    res.json({ tools });
});

/**
 * Execute tool
 */
app.post('/call', (req, res) => {
    const { toolName, arguments } = req.body;

    try {
        switch (toolName) {
            case 'createTask':
                return handleCreateTask(arguments, res);
            case 'listTasks':
                return handleListTasks(res);
            case 'updateTask':
                return handleUpdateTask(arguments, res);
            default:
                return res.status(400).json({ error: 'Unknown tool' });
        }
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

/**
 * Tool implementations
 */
function handleCreateTask(args, res) {
    // Connect to database, create task
    const task = {
        id: 'task-' + Date.now(),
        title: args.title,
        description: args.description || '',
        priority: args.priority || 'medium',
        status: 'open',
        createdAt: new Date()
    };

    // Save to database
    console.log('Task created:', task);

    res.json({
        status: 'success',
        data: task
    });
}

function handleListTasks(res) {
    // Query database for all tasks
    const tasks = [
        { id: 'task-1', title: 'Task 1', status: 'open' },
        { id: 'task-2', title: 'Task 2', status: 'closed' }
    ];

    res.json({
        status: 'success',
        data: tasks
    });
}

function handleUpdateTask(args, res) {
    // Update task in database
    const updated = {
        id: args.taskId,
        status: args.status,
        updatedAt: new Date()
    };

    res.json({
        status: 'success',
        data: updated
    });
}

// Start server
app.listen(3000, () => {
    console.log('Custom MCP Server running on port 3000');
});
```

## Production Features

### **Error Handling Strategy**

```java
/**
 * Production-grade error handling
 */
@Service
public class ProductionChatService {

    private static final Logger logger = 
        LoggerFactory.getLogger(ProductionChatService.class);

    private final ChatClient chatClient;

    /**
     * Chat with comprehensive error handling
     */
    public ChatResponse chat(String userMessage) {
        try {
            // Validate input
            if (userMessage == null || userMessage.trim().isEmpty()) {
                throw new IllegalArgumentException("Message cannot be empty");
            }

            logger.info("Processing user message: {}", userMessage);

            // Invoke LLM with tools
            String response = chatClient.prompt()
                .user(userMessage)
                .call()
                .content();

            logger.info("Chat completed successfully");
            return new ChatResponse(response);

        } catch (IllegalArgumentException e) {
            logger.warn("Invalid input: {}", e.getMessage());
            return new ChatResponse("Error: " + e.getMessage());

        } catch (HttpServerErrorException e) {
            logger.error("LLM service error", e);
            return new ChatResponse("Service temporarily unavailable. Please try again.");

        } catch (TimeoutException e) {
            logger.error("LLM request timeout", e);
            return new ChatResponse("Request timed out. Please try with a shorter message.");

        } catch (Exception e) {
            logger.error("Unexpected error in chat", e);
            return new ChatResponse("An unexpected error occurred. Please contact support.");
        }
    }
}
```

### **Comprehensive Logging**

```java
/**
 * Structured logging for production
 */
@Aspect
@Component
public class ChatLoggingAspect {

    private static final Logger logger = 
        LoggerFactory.getLogger(ChatLoggingAspect.class);

    @Around("@annotation(LogChat)")
    public Object logChatOperation(ProceedingJoinPoint pjp) throws Throwable {
        String methodName = pjp.getSignature().getName();
        long startTime = System.currentTimeMillis();

        try {
            logger.info("Chat operation started: {}", methodName);
            Object result = pjp.proceed();
            long duration = System.currentTimeMillis() - startTime;
            logger.info("Chat operation completed: {} ({}ms)", methodName, duration);
            return result;

        } catch (Exception e) {
            long duration = System.currentTimeMillis() - startTime;
            logger.error("Chat operation failed: {} ({}ms): {}", 
                        methodName, duration, e.getMessage());
            throw e;
        }
    }
}

@Retention(RetentionPolicy.RUNTIME)
@Target(ElementType.METHOD)
public @interface LogChat {
}
```

### **Monitoring & Metrics**

```java
/**
 * Metrics collection for monitoring
 */
@Service
public class ChatMetricsService {

    private final MeterRegistry meterRegistry;

    public ChatMetricsService(MeterRegistry meterRegistry) {
        this.meterRegistry = meterRegistry;
    }

    public void recordChatRequest(String toolName, long duration) {
        // Track requests per tool
        meterRegistry.counter("chat.requests.total",
            "tool", toolName).increment();

        // Track response time
        meterRegistry.timer("chat.response.time",
            "tool", toolName).record(duration, TimeUnit.MILLISECONDS);
    }

    public void recordChatError(String errorType) {
        meterRegistry.counter("chat.errors.total",
            "error_type", errorType).increment();
    }
}
```

### **Resource Management**

```java
/**
 * Manage resource limits for multiple servers
 */
@Service
public class ResourceManagementService {

    private static final Logger logger = 
        LoggerFactory.getLogger(ResourceManagementService.class);

    private final RateLimiter rateLimiter;

    public ResourceManagementService() {
        // 100 requests per second limit
        this.rateLimiter = RateLimiter.create(100);
    }

    /**
     * Execute tool with resource limits
     */
    public String executeTool(String toolName, Map<String, Object> params) {
        // Check rate limit
        if (!rateLimiter.tryAcquire()) {
            logger.warn("Rate limit exceeded for tool: {}", toolName);
            throw new RateLimitExceededException("Too many requests");
        }

        // Check timeout
        try {
            return executeWithTimeout(toolName, params, 30, TimeUnit.SECONDS);
        } catch (TimeoutException e) {
            logger.error("Tool execution timeout: {}", toolName);
            throw new ToolExecutionException("Tool execution timed out");
        }
    }

    private String executeWithTimeout(String toolName, Map<String, Object> params,
                                      long timeout, TimeUnit unit)
            throws TimeoutException {
        // Execute with timeout
        return CompletableFuture.supplyAsync(
            () -> executeTool(toolName, params)
        ).get(timeout, unit);
    }
}
```
